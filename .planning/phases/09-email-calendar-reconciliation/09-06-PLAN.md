    import type { EmailAccountWithAI } from "@/utils/llms/types";
    import type { Logger } from "@/utils/logger";
    import prisma from "@/utils/prisma";
    import { getUpcomingEvents } from "@/utils/calendar/upcoming-events";
    import { convertEmailHtmlToText } from "@/utils/mail"; // verify exact import — see RESEARCH §F
    import { extractFromIcs } from "./ics-path";
    import { extractCandidateEvent } from "./extract";
    import { decideOutcome } from "./match";
    import { eventSignature } from "./signature";
    import {
      createReconciliationRecord,
      findExistingReconciliationRecord,
      findStalePendingRecord,
      updateReconciliationRecord,
    } from "./persist";
    import { createCalendarEvent } from "./create-event";

    // (matchesKeywordBackstop + CALENDAR_KEYWORDS already defined in Task 1, above)

    async function isClassifiedAsCalendar({
      emailAccountId,
      messageId,
    }: {
      emailAccountId: string;
      messageId: string;
    }): Promise<boolean> {
      const executed = await prisma.executedRule.findFirst({
        where: { emailAccountId, messageId, rule: { systemType: "CALENDAR" } },
        select: { id: true },
      });
      return executed !== null;
    }

    function truncateBody(parsedMessage: ParsedMessage): string {
      const raw =
        parsedMessage.textPlain ??
        convertEmailHtmlToText(parsedMessage.textHtml ?? "");
      return raw.slice(0, 2000);
    }

    export async function reconcileMessage({
      parsedMessage,
      emailAccount,
      emailAccountId,
      logger,
    }: {
      parsedMessage: ParsedMessage;
      emailAccount: EmailAccountWithAI;
      emailAccountId: string;
      logger: Logger;
    }): Promise<void> {
      const messageId = parsedMessage.id;
      const threadId = parsedMessage.threadId;
      const senderEmail = parsedMessage.headers?.from ?? "";
      const subject = parsedMessage.headers?.subject ?? "";

      try {
        // 1. Pre-filter (D-01)
        const icsCandidate = extractFromIcs(parsedMessage);
        let pathA = icsCandidate !== null;
        let pathB = false;
        let bodyTruncated = "";

        if (!pathA) {
          const isCalendar = await isClassifiedAsCalendar({ emailAccountId, messageId });
          bodyTruncated = truncateBody(parsedMessage);
          pathB = isCalendar || matchesKeywordBackstop({ subject, body: bodyTruncated });
          if (!pathB) return;
        }

        // 2. Idempotency fast-path
        const existing = await findExistingReconciliationRecord({ emailAccountId, messageId });
        let existingRowForReuse: { id: string } | null = null;
        if (existing) {
          const stale = await findStalePendingRecord({ emailAccountId, messageId });
          if (!stale) {
            // Non-stale row exists → idempotency no-op
            logger.info("Reconciliation already processed (idempotency hit)", {
              emailAccountId,
              messageId,
              outcome: existing.outcome,
            });
            return;
          }
          // Stale-PENDING (D-16) → reuse the existing row id, skip create, proceed past extraction
          existingRowForReuse = { id: stale.id };
        }

        // 3. Extract
        const candidate = pathA
          ? icsCandidate!
          : await extractCandidateEvent({
              email: { subject, from: senderEmail, bodyTruncated },
              emailAccount,
              logger,
            });

        // Heuristic for isAllDay — endISO null AND startISO time is midnight is the standard signal;
        // for v1 default to !startISO.includes("T") OR (endISO && endISO === nextDay-of-start)
        const isAllDay = (() => {
          if (!candidate.startISO) return false;
          // simple heuristic: presence of explicit time portion
          return /T00:00:00/.test(candidate.startISO) && candidate.endISO === null;
        })();

        const sig = eventSignature(candidate.title, candidate.startISO);

        // 4. Create-or-reuse persistence row
        let recordId: string;
        if (existingRowForReuse) {
          recordId = existingRowForReuse.id;
        } else {
          const created = await createReconciliationRecord({
            input: {
              emailAccountId,
              messageId,
              threadId,
              eventSignature: sig,
              extractedTitle: candidate.title,
              extractedStart: candidate.startISO ? new Date(candidate.startISO) : new Date(0),
              extractedEnd: candidate.endISO ? new Date(candidate.endISO) : null,
              extractedLocation: candidate.location,
              extractedAttendees: candidate.attendees,
              candidateConfidence: candidate.confidence,
            },
            logger,
          });
          if (!created.created) {
            // P2002 — another concurrent webhook landed first; no-op
            return;
          }
          recordId = created.record.id;
        }

        // 5. Match
        const now = new Date();
        const upcoming = await getUpcomingEvents({ emailAccountId, now, logger }).catch(() => []);
        const { outcome, matchedEventId } = decideOutcome({
          candidate: {
            title: candidate.title,
            startISO: candidate.startISO,
            isAllDay,
          },
          existingEvents: upcoming,
        });

        // 6. Update with outcome (and matched event id for MATCHED/AMBIGUOUS)
        if (outcome !== "CREATED") {
          await updateReconciliationRecord({
            id: recordId,
            data: { outcome, googleEventId: matchedEventId },
          });
          return;
        }

        // 7. CREATED path → call Google
        const tz = emailAccount.timezone ?? "America/New_York";
        const inserted = await createCalendarEvent({
          input: {
            emailAccountId,
            messageId,
            threadId,
            senderEmail,
            timezone: tz,
            candidate: {
              title: candidate.title,
              startISO: candidate.startISO,
              endISO: candidate.endISO,
              location: candidate.location,
              isAllDay,
            },
          },
          logger,
        });
        if (inserted.ok) {
          await updateReconciliationRecord({
            id: recordId,
            data: {
              outcome: "CREATED",
              googleEventId: inserted.googleEventId,
              googleEventHtmlLink: inserted.googleEventHtmlLink,
            },
          });
        } else {
          await updateReconciliationRecord({
            id: recordId,
            data: { outcome: "FAILED", errorMessage: inserted.reason },
          });
        }
      } catch (error) {
        logger.error("Reconciliation failed", {
          emailAccountId,
          messageId,
          error,
        });
        // Best-effort: try to mark any partially-written record as FAILED
        try {
          const existing = await findExistingReconciliationRecord({ emailAccountId, messageId });
          if (existing) {
            await updateReconciliationRecord({
              id: existing.id,
              data: {
                outcome: "FAILED",
                errorMessage: (error instanceof Error ? error.message : String(error)).slice(0, 200),
              },
            });
          }
        } catch {
          // swallow — failure isolation (OPS-01)
        }
        // DO NOT rethrow.
      }
    }
