D-07 algorithm (RESEARCH §6 Option B — recommended):
```ts
export function titleSimilarity(a: string, b: string): number {
  const tokens = (s: string) => s.toLowerCase().trim().split(/\s+/).filter(Boolean);
  const A = new Set(tokens(a));
  const B = new Set(tokens(b));
  if (A.size === 0 && B.size === 0) return 1;
  if (A.size === 0 || B.size === 0) return 0;
  let intersection = 0;
  for (const t of A) if (B.has(t)) intersection++;
  return (2 * intersection) / (A.size + B.size);
}
```

D-08 all-day rule: For all-day candidates, compare by date string `YYYY-MM-DD` equality, not by time window.

eventSignature contract (D-13 + RESEARCH §E):
```ts
import { createHash } from "node:crypto";
export function normalizeTitle(title: string): string {
  return title.toLowerCase().trim().replace(/\s+/g, " ");
}
export function eventSignature(title: string, startISO: string): string {
  return createHash("sha256").update(`${normalizeTitle(title)}|${startISO}`).digest("hex");
}
```
This contract is IMMUTABLE post-merge — D-14 unique constraint on existing rows depends on hash stability.

    ```ts
    import type { NormalizedCalendarEvent } from "@/utils/calendar/upcoming-events-types";
    import { titleSimilarity } from "./dice";

    export type ReconcileOutcome = "MATCHED" | "CREATED" | "AMBIGUOUS";

    const STRONG_SIM = 0.7;
    const WEAK_SIM = 0.4;
    const TIME_WINDOW_MS = 60 * 60 * 1000;

    export function decideOutcome({
      candidate,
      existingEvents,
    }: {
      candidate: { title: string; startISO: string; isAllDay: boolean };
      existingEvents: NormalizedCalendarEvent[];
    }): { outcome: ReconcileOutcome; matchedEventId: string | null } {
      if (candidate.isAllDay) {
        const candDate = candidate.startISO.slice(0, 10);
        for (const e of existingEvents) {
          const eDate = e.start?.date ?? e.start?.dateTime?.slice(0, 10) ?? "";
          const sim = titleSimilarity(candidate.title, e.summary ?? "");
          if (eDate === candDate && sim >= STRONG_SIM) {
            return { outcome: "MATCHED", matchedEventId: e.id };
          }
        }
        return { outcome: "CREATED", matchedEventId: null };
      }

      const candMs = Date.parse(candidate.startISO);

      // Step 1: strong title sim AND time within ±60min → MATCHED
      for (const e of existingEvents) {
        const eIso = e.start?.dateTime ?? null;
        if (!eIso) continue;
        const sim = titleSimilarity(candidate.title, e.summary ?? "");
        const diff = Math.abs(Date.parse(eIso) - candMs);
        if (sim >= STRONG_SIM && diff <= TIME_WINDOW_MS) {
          return { outcome: "MATCHED", matchedEventId: e.id };
        }
      }

      // Step 2: strong title sim anywhere in window, time > 60min → AMBIGUOUS (reschedule)
      for (const e of existingEvents) {
        const sim = titleSimilarity(candidate.title, e.summary ?? "");
        if (sim >= STRONG_SIM) {
          return { outcome: "AMBIGUOUS", matchedEventId: e.id };
        }
      }

      // Step 3: same-day + weak sim → AMBIGUOUS (near-match)
      const candDate = candidate.startISO.slice(0, 10);
      for (const e of existingEvents) {
        const eDate = e.start?.dateTime?.slice(0, 10) ?? e.start?.date ?? "";
        const sim = titleSimilarity(candidate.title, e.summary ?? "");
        if (eDate === candDate && sim >= WEAK_SIM && sim < STRONG_SIM) {
          return { outcome: "AMBIGUOUS", matchedEventId: e.id };
        }
      }

      return { outcome: "CREATED", matchedEventId: null };
    }
    ```

    Re-run. All 7 tests pass.
