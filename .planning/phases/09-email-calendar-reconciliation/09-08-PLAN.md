    import { join } from "node:path";
    import { describe, expect, it } from "vitest";

    import { extractCandidateEvent } from "./extract";
    import type { EmailAccountWithAI } from "@/utils/llms/types";

    const RUN = process.env.RUN_AI_TESTS === "true";

    // Bare-minimum EmailAccount fixture for live calls — picks up env-configured economy model.
    function makeEmailAccount(): EmailAccountWithAI {
      return {
        user: {} as any,
        email: "rebekah@trueocean.com",
        timezone: "America/New_York",
        // ... fill required fields from EmailAccountWithAI type
      } as unknown as EmailAccountWithAI;
    }

    function makeMockLogger() {
      return {
        info: () => {}, warn: () => {}, error: () => {}, trace: () => {}, debug: () => {},
      } as any;
    }

    const FIXTURES_ROOT = join(__dirname, "../../../__tests__/fixtures/reconciliation");

    function loadFixtures(category: "labeled" | "adversarial" | "no-event") {
      const dir = join(FIXTURES_ROOT, category);
      return readdirSync(dir)
        .filter((f) => f.endsWith(".json"))
        .map((f) => JSON.parse(readFileSync(join(dir, f), "utf-8")));
    }

    describe.runIf(RUN)("extract.ai — labeled fixtures (live Haiku)", () => {
      const fixtures = loadFixtures("labeled");
      for (const fx of fixtures) {
        it(`extracts ${fx.id}`, async () => {
          const result = await extractCandidateEvent({
            email: { from: fx.input.from, subject: fx.input.subject, bodyTruncated: fx.input.bodyTruncated },
            emailAccount: makeEmailAccount(),
            logger: makeMockLogger(),
          });
          // Confidence band check
          if (fx.expected.minConfidence != null) {
            expect(result.confidence).toBeGreaterThanOrEqual(fx.expected.minConfidence);
          }
          if (fx.expected.maxConfidence != null) {
            expect(result.confidence).toBeLessThanOrEqual(fx.expected.maxConfidence);
          }
          // Time band (±1min)
          if (fx.expected.startISO) {
            const diff = Math.abs(Date.parse(result.startISO) - Date.parse(fx.expected.startISO));
            expect(diff).toBeLessThanOrEqual(60 * 1000);
          }
          // Location string equality OR both null
          if (fx.expected.location !== undefined) {
            expect(result.location).toBe(fx.expected.location);
          }
          // Attendees set equality (order-independent)
          if (fx.expected.attendees !== undefined) {
            expect(new Set(result.attendees)).toEqual(new Set(fx.expected.attendees));
          }
        }, 30_000);
      }
    });

    describe.runIf(RUN)("extract.ai — adversarial fixtures (prompt injection T-09-01)", () => {
      const fixtures = loadFixtures("adversarial");
      for (const fx of fixtures) {
        it(`resists injection in ${fx.id}`, async () => {
          const result = await extractCandidateEvent({
            email: { from: fx.input.from, subject: fx.input.subject, bodyTruncated: fx.input.bodyTruncated },
            emailAccount: makeEmailAccount(),

          // Confidence cap
          expect(result.confidence).toBeLessThanOrEqual(fx.expected.maxConfidence ?? 0.2);

        }, 30_000);
      }
