  <behavior>
    Test file structure:

    ```ts
    const RUN = process.env.RUN_AI_TESTS === "true";

    describe.runIf(RUN)("arbitrate.ai.test (live)", () => {
      const fixtures = readdirSync(...).map(f => JSON.parse(readFileSync(...)));

      for (const fx of fixtures) {
        it(`fixture ${fx.id}: ${fx.notes ?? ''}`, async () => {
          // Step 1 — overlap pre-check (validates 11-02 interval helper too)
          const overlaps = findIntervalOverlaps({
            candidateStartISO: fx.candidate.startISO,
            candidateEndISO: fx.candidate.endISO,
            existingEvents: fx.daySchedule,
          });
          if (!fx.expectedArbitration.shouldCallArbiter) {
            expect(overlaps).toHaveLength(0);
            return; // no arbiter call expected; pass.
          }
          expect(overlaps.length).toBeGreaterThan(0);

          // Step 2 — live arbiter call
          const result = await arbitrateOverlap({
            email: fx.input,
            candidate: fx.candidate,
            daySchedule: fx.daySchedule,
            emailAccount: makeEmailAccount(),
            logger: makeMockLogger(),
          });

          expect(result.verdict).toBe(fx.expectedArbitration.verdict);
          if (fx.expectedArbitration.verdict === 'SAME' || fx.expectedArbitration.verdict === 'RESCHEDULE') {
            const expectedId = fx.daySchedule.find(e => e.id === fx.expectedArbitration.matchedEventIdRef)?.id;
            expect(result.matchedEventId).toBe(expectedId);
          } else {
            expect(result.matchedEventId).toBeNull();
          }
        });
      }
    });
    ```
