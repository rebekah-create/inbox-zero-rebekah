    Where `runArbitrationOrFallback` is a local helper (defined inside the file, NOT exported) that wraps the arbitration call with D-08 fallback:

    ```
    async function runArbitrationOrFallback(args): Promise<{
      outcome: ReconciliationOutcome;
      matchedEventId: string | null;
      rescheduleOfEventId: string | null;
      arbiterErrorMessage: string | null;
    }> {
      try {
        const arb = await arbitrateOverlap({...});
        switch (arb.verdict) {
          case 'SAME':
            return { outcome: 'MATCHED', matchedEventId: arb.matchedEventId, rescheduleOfEventId: null, arbiterErrorMessage: null };
          case 'SEPARATE':
            return { outcome: 'CREATED', matchedEventId: null, rescheduleOfEventId: null, arbiterErrorMessage: null };
          case 'RESCHEDULE':
            // Stage the linkage. Actual Google calls happen in step 6/7.
            return { outcome: 'RESCHEDULE', matchedEventId: null, rescheduleOfEventId: arb.matchedEventId, arbiterErrorMessage: null };
          case 'SKIP':
            return { outcome: 'FAILED', matchedEventId: null, rescheduleOfEventId: null, arbiterErrorMessage: 'arbiter_skip' };
        }
      } catch (err) {
        logger.warn('Reconciliation arbitration failed; falling through to CREATE', {
          emailAccountId: args.emailAccountId,
          messageId: args.messageId,
          error: err instanceof Error ? err.message : String(err),
        });
        return { outcome: 'CREATED', matchedEventId: null, rescheduleOfEventId: null, arbiterErrorMessage: null };
      }
    }
    ```
