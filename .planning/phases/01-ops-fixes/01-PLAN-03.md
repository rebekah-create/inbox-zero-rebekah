```typescript
import { describe, it, expect, vi, beforeEach } from "vitest";

// Mock the env module before importing the module under test.
// Adjust the mock path below to match the exact import used in auth-signup-policy.ts.
vi.mock("../../../env", () => ({
  env: {
    AUTH_ALLOWED_EMAIL_DOMAINS: undefined,
    AUTH_ALLOWED_EMAILS: undefined,
  },
}));

import { isAllowedAuthSignupEmail } from "./auth-signup-policy";
import { env } from "../../../env";

describe("isAllowedAuthSignupEmail", () => {
  beforeEach(() => {
    vi.resetAllMocks();
    // Default: no restrictions set (open signup state)
    (env as any).AUTH_ALLOWED_EMAIL_DOMAINS = undefined;
    (env as any).AUTH_ALLOWED_EMAILS = undefined;
  });

  it("blocks a non-trueocean.com address when AUTH_ALLOWED_EMAIL_DOMAINS=trueocean.com", async () => {
    (env as any).AUTH_ALLOWED_EMAIL_DOMAINS = ["trueocean.com"];
    await expect(
      isAllowedAuthSignupEmail("attacker@gmail.com")
    ).rejects.toMatchObject({ code: "signup_not_allowed" });
  });

  it("allows a trueocean.com address when AUTH_ALLOWED_EMAIL_DOMAINS=trueocean.com", async () => {
    (env as any).AUTH_ALLOWED_EMAIL_DOMAINS = ["trueocean.com"];
    await expect(
      isAllowedAuthSignupEmail("rebekah@trueocean.com")
    ).resolves.toBeUndefined();
  });

  it("allows any address when no restrictions are configured (open signup)", async () => {
    await expect(
      isAllowedAuthSignupEmail("anyone@anywhere.com")
    ).resolves.toBeUndefined();
  });
});
```

| AWS SSM → Server .env | Secrets loaded at boot via load-secrets.sh; SSM is the authoritative source |

## STRIDE Threat Register

| Threat ID | Category | Component | Disposition | Mitigation Plan |
|-----------|----------|-----------|-------------|-----------------|
| T-03-01 | Elevation of Privilege | Signup endpoint open to all emails | mitigate | AUTH_ALLOWED_EMAIL_DOMAINS=trueocean.com added to SSM; isAllowedAuthSignupEmail blocks all non-trueocean.com addresses after env var is loaded |
| T-03-02 | Elevation of Privilege | Missing env var leaves signup open | mitigate | End-to-end verification (D-06) required — Task 3 confirms blocked signup with a live test, not just grep of .env |
| T-03-03 | Spoofing | Digest email sent from wrong domain | mitigate | RESEND_FROM_EMAIL explicitly set in SSM to "Inbox Zero <inbox-digest@tdfurn.com>"; live test confirms received email From header |
| T-03-04 | Tampering | CRON_SECRET not set (digest endpoint unprotected) | accept | CRON_SECRET is already in SSM from prior setup (confirmed via Step 1 parameter audit); cron endpoint requires Bearer token |
| T-03-05 | Information Disclosure | Secrets visible in SSM audit log | accept | SSM SecureString parameters are encrypted at rest; IAM access controls govern who can read them — acceptable for single-user personal system |
